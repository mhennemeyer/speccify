    def method_missing(name, *args, &block)
      if (name.to_s =~ /^be_(.+)/)
        Speccify::Functions::build_matcher(name, args) do |given, matcher, args|
          given.send(($1 + "?").to_sym)
        end
      else
        raise NoMethodError.new(name.to_s)
      end
    end
